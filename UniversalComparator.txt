public class UniversalComparator {
	
	public static Integer compare(Object val1, Object val2) {
	Decimal convertedVar1;
	Decimal convertedVar2;

		try{
			convertedVar1 = Decimal.valueOf(String.valueOf(val1));
			convertedVar2 = Decimal.valueOf(String.valueOf(val2));
		} catch(Exception e) {
			
		}

		if (convertedVar1 < convertedVar2) {
			return -1;
		} 
		if (convertedVar1 > convertedVar2) {
			return 1;
		} 
		return 0;
	}
}
